#| include: false
library(dplyr)
library(tidyr)
library(lubridate)
library(stringr)
library(plotly)
library(reactable)
library(DT)
library(htmlwidgets)
library(crosstalk)
library(htmltools)

# Custom color palette
custom_colors <- list(
  primary = "#2E8B57",
  secondary = "#20B2AA", 
  success = "#28a745",
  warning = "#ffc107",
  danger = "#dc3545",
  info = "#17a2b8",
  light = "#f8f9fa",
  dark = "#343a40",
  gradient = c("#2E8B57", "#20B2AA", "#66CDAA", "#98FB98", "#90EE90")
)

# Helper functions
format_percent <- function(x, digits = 1) {
  paste0(round(x * 100, digits), "%")
}

format_number <- function(x, digits = 0) {
  format(round(x, digits), big.mark = ",", scientific = FALSE)
}

# Load and process data
df_raw <- readRDS("myapp/df.rds")

df_processed <- df_raw %>%
  mutate(across(everything(), ~ na_if(., "---"))) %>%
  mutate(across(where(is.list), ~ sapply(., function(x) if(length(x) > 0) x[1] else NA))) %>%
  mutate(
    completed_time = ymd_hms(completed_time),
    started_time = ymd_hms(started_time)
  ) %>%
  mutate(
    date = as.Date(completed_time),
    week = floor_date(date, "week"),
    month = floor_date(date, "month"),
    hour_started = hour(started_time),
    day_of_week = wday(date, label = TRUE, week_start = 1),
    is_weekend = day_of_week %in% c("Sat", "Sun"),
    duration_minutes = as.numeric(difftime(completed_time, started_time, units = "mins")),
    is_night_survey = hour_started >= 19 | hour_started < 6,
    is_short_survey = duration_minutes <= 5,
    is_long_survey = duration_minutes >= 60
  ) %>%
  mutate(across(c(starts_with("ps_num_planted_"), starts_with("num_surv_")), as.numeric))

df_completed <- df_processed %>% filter(consent == 1)

# Site statistics
site_stats <- df_completed %>%
  group_by(site) %>%
  summarise(
    total_surveys = n(),
    avg_duration = round(mean(duration_minutes, na.rm = TRUE), 1),
    completion_rate = round((n() / nrow(df_completed)) * 100, 1),
    unique_enumerators = n_distinct(username),
    avg_hh_size = round(mean(hh_size, na.rm = TRUE), 1),
    total_planted = sum(across(starts_with("ps_num_planted_")), na.rm = TRUE),
    total_survived = sum(across(starts_with("num_surv_")), na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    survival_rate = ifelse(total_planted > 0, (total_survived / total_planted) * 100, 0)
  ) %>%
  arrange(desc(total_surveys))

# Create shared data for crosstalk
shared_site_data <- SharedData$new(site_stats)

#| echo: false

# Create rankings for different metrics
site_rankings <- site_stats %>%
  mutate(
    survey_rank = rank(-total_surveys),
    survival_rank = rank(-survival_rate),
    duration_rank = rank(avg_duration),  # Lower duration is better
    overall_score = (survey_rank + survival_rank + duration_rank) / 3
  ) %>%
  arrange(overall_score) %>%
  select(site, survey_rank, survival_rank, duration_rank, overall_score) %>%
  head(10)

# Reshape for plotting
rankings_long <- site_rankings %>%
  pivot_longer(cols = c(survey_rank, survival_rank, duration_rank), 
               names_to = "metric", values_to = "rank") %>%
  mutate(
    metric = case_when(
      metric == "survey_rank" ~ "Survey Count",
      metric == "survival_rank" ~ "Survival Rate", 
      metric == "duration_rank" ~ "Duration (Lower=Better)"
    )
  )

p5 <- plot_ly(rankings_long, 
        x = ~metric, 
        y = ~rank, 
        color = ~site,
        type = 'scatter',
        mode = 'lines+markers',
        line = list(width = 3),
        marker = list(size = 8),
        hovertemplate = "Site: %{fullData.name}<br>Metric: %{x}<br>Rank: %{y}<extra></extra>") %>%
  layout(
    title = "Site Rankings Across Key Metrics (Lower Rank = Better Performance)",
    xaxis = list(title = "Performance Metric"),
    yaxis = list(title = "Rank", autorange = "reversed"),
    paper_bgcolor = "transparent", 
    plot_bgcolor = "transparent",
    legend = list(orientation = 'h', y = 1.15, x = 0.5, xanchor = 'center')
  )

# Calculate efficiency metrics
efficiency_data <- site_stats %>%
  mutate(
    surveys_per_enumerator = total_surveys / unique_enumerators,
    trees_per_survey = total_planted / total_surveys,
    efficiency_score = (surveys_per_enumerator * survival_rate) / avg_duration
  ) %>%
  arrange(desc(efficiency_score)) %>%
  head(10)

p6 <- plot_ly(efficiency_data, 
        x = ~reorder(site, efficiency_score), 
        y = ~efficiency_score, 
        type = 'bar',
        marker = list(color = custom_colors$info),
